#Don't want to include the blank in the averages for each group

Exp1_10min_noblanks <- Exp1_10min %>%
  filter(!str_trim(Replicate) %in% c("Cell Blank", "Buffer Blank"))

#Also need to normalize the data and subtract the value for average blanks
  #Finding the average absorbance
blank_absorbance<- Exp1_10min %>% 
  filter(grepl("Blank", Replicate, ignore.case=TRUE))

mean_blank<-mean(blank_absorbance$Absorbance_420)
  #subtracting the average absorbance from all the wells in my experimental data

Exp1_10min_normalized<- Exp1_10min_noblanks %>% 
  mutate(Absorbance_420_normalized=Absorbance_420-mean_blank)

GET SUMMARY STATISTICS

#miller units= 1000 x OD(420 nm)/ Time (min) x Culture Volume (mL) x OD(600 nm)
#each culture was normalized to the same (V × A 600)...For Experiment 1 it was 800uL*0.357 OD600
#But I only used 100uL of cell lysate?
#so Miller unit will be A420/(10*.1*0.357) or divide everything by 0.357

Exp1_10min_summary<- Exp1_10min_normalized %>% 
  mutate(millerunits=1000*Absorbance_420_normalized/0.357) %>% 
  group_by(Group) %>% 
  summarise(mean_absorbance=mean(Absorbance_420_normalized), sd_absorbance=sd(Absorbance_420_normalized), mean_millerunits=mean(millerunits), sd_millerunits=sd(millerunits))

#I want to compare the expression of the constructs across the different backgrounds
  #Need to first modify the data so I can sort by the different constructs
        #OK I actually didn't really need to do this that much
Exp1_10min_grouped<-Exp1_10min_normalized %>% 
  mutate(Group = recode(Group, "1" = "\u0394SatS + Complement + pGP1", 
                  "2"="\u0394SatS + Complement + pGP2",
                  "3"="WT + pYUB76",
                  "4"="WT + pGP1",
                  "5"="WT + pGP2",
                  "6"= "\u0394SatS + pGP1",
                  "7"="\u0394SatS + pGP2")) %>% 
 mutate(Construct = ifelse(grepl("pGP1", Group), "pGP1",
                            ifelse(grepl("pGP2", Group), "pGP2",
                                   ifelse(grepl("pYUB76", Group), "Empty", "Other"))))

  #OK now I can filter out pGP1 or 2, which will allow me to compare fold change and plot with just the 1 experimental construct

  #Except I want to divide by the average absorbance of the construct in the wild type...perhaps I should stick to using just the summary stats to graph? I will have to give up on the idea of a faceted graph :(

Exp1_10min_pGP1_summary<- Exp1_10min_grouped %>% 
  filter(!Construct=="pGP2") %>% 
  group_by(Group) %>% 
  summarize(mean_absorbance_420=mean(Absorbance_420_normalized),
            sd_absorbance_420=sd(Absorbance_420_normalized))

###Remember to check here when using for other experiments that its taking the right value to measure fold change

WT_pGP1_A420<- Exp1_10min_pGP1_summary$mean_absorbance_420[1]

  #So NOW I can mutate a column with the fold change from WT and graph that
Exp1_10min_pGP1_summary %>% 
  mutate(fold_change_WT=mean_absorbance_420/WT_pGP1_A420) %>% 
 ggplot(aes(x=Group, y=fold_change_WT, fill=Group))+
  geom_bar(stat="identity")+
  geom_errorbar(aes(ymin = fold_change_WT - sd_absorbance_420 / WT_pGP1_A420, 
                    ymax = fold_change_WT + sd_absorbance_420 / WT_pGP1_A420),
                width = 0.1,  #unsure if this is actually the right way to do error bars for fold change
                position = position_dodge(width = 0.9))+
  labs(title = bquote("Change in LacZ Expression of pGP1 Construct in"~italic("M. smegmatis")),
       x = bquote(italic("M. smegmatis")~"Background + LacZ Construct"),
       y=bquote("Fold Change from WT"))+
  theme_few()+
  scale_fill_brewer(palette = "Set2")+
  theme(axis.text.x = element_text(angle = 45, hjust = 1))
